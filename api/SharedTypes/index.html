interface HammerConfig {
Damage: number--

The amount of damage the hammer does to a block each hit

MaxDistance: number--

The maximum distance the hammer can perform an action

PlaceRPM: number--

The rate at which the hammer can place blocks per minute

DeleteRPM: number--

The rate at which the hammer hits blocks per minute

}
